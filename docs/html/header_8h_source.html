160  double wrapk (double k)
161  {
162  double wpl = w+cosp*k*v;
163  double resk;
164

167  }

173  }

175  resk = resk/(1.-exp(-beta*hbar*wpl));
176  }

179  }

183  pre = pow(cosp,2);
184  }
185  if (sig == 1) {

187  }

190  }

192  lim2 = kcut/(2*za);
193  caseT = 0;

195  if (cosp < 0) {

198  caseT = 1;

203  resphi = pre*integ(wrapk, lim1, lim2, relerr*recerr*recerr);

215 // ... the (2,2) component with sin^2(phi) as prefactor

218 // and the (2,2) component as sin^2 = 1 - cos^2
